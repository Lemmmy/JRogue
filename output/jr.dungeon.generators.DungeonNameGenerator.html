	public static String generate() {
		return StringReplacer.replace(
			RandomUtils.randomFrom(NAME_TEMPLATES),
			Pattern.compile("\\{(\\w+)}"),
			(Matcher m) -> RandomUtils.randomFrom(NAME_OBJECTS.get(m.group(1)))
		);
	}
